--[1-3]공지사항 상제보기
ALTER Proc [dbo].[UP_ViewSCM_Order]
   @SCM_NoticeID Int
AS
   Set Nocount on
   --조회수증가
   Update 
      SCM_Order
   Set
      ReadCount = ReadCount + 1
   Where
      SCM_NoticeID = @SCM_NoticeID
   --번호값에 해당하는 정보 읽어오기
   
    
   Select 
      *
   From
      SCM_Order
   Where
      SCM_NoticeID = @SCM_NoticeID
