ALTER Proc [dbo].[UP_ListSCM_Order]
   @code1 varchar(2)
AS   
   Select 
      * 
      , case  when Progress1='종료' then 'red' else 'black' end as color
   From
      SCM_Order
      where code1 like @code1+'%'
   Order by 
     color,dead1,SCM_NoticeID Desc
